using namespace std;

//skaicio generavimas
int genAtsitiktinis() {
    return rand() % 100 + 1;
}

// zaidimo funkcija
void zaidimas() {
    int atsitiktinisSkaicius = genAtsitiktinis();
    int spejimas;
    cout << "Atspekite skaiciu nuo 1 iki 100:" << endl;
    while (true) {
        cout << "Jusu spejimas: ";
        cin >> spejimas;

        if (spejimas > atsitiktinisSkaicius) {
            cout << "Per didelis!" << endl;
        } else if (spejimas < atsitiktinisSkaicius) {
            cout << "Per mazas!" << endl;
        } else {
            cout << "Atspejote teisingai!" << endl;
            break;
        }
    }
}
//fizzbuzz funkcija
void fizzbuzz(int n) {
    for (int i = 1; i <= n; i++) {
        if (i % 3 == 0 && i % 5 == 0) {
            cout << "FizzBuzz" << endl;
        } else if (i % 3 == 0) {
            cout << "Fizz" << endl;
        } else if (i % 5 == 0) {
            cout << "Buzz" << endl;
        } else {
            cout << i << endl;
        }
    }
}
